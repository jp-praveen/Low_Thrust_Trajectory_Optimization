1function [t_out,y_out] = ode45_fast(tspan,y0,c, Thr, SI2CAN, kd) %#codegen
  2global atol rtol pow A B E
  3
  4t0 = tspan(1);
  5tfinal = tspan(2);
  6f0 = StCO_Dynamics_MEE(t0,y0, c, Thr, SI2CAN, kd);                              %<--------------------------
  7threshold = atol / rtol;
  8
  9% By default, hmax is 1/10 of the interval.
 10hmax = 0.1*(tfinal-t0);
 11tnew = 0;
 12ynew = y0;
 13t = t0;
 14y = y0;
 15Neq = 14;                                                                  %<--------------------------
 16f = zeros(Neq,7,'double'); % the first number is number of equations
 17hmin = 16*eps(t);                                                          % Check made to 14 from 16
 18htspan = tfinal-t0;
 19
 20% Compute an initial step size h using y'(t).
 21absh = min(hmax, htspan);
 22rh = norm(f0 ./ max(abs(y),threshold),inf) / (0.8 * rtol^pow);
 23
 24if absh * rh > 1
 25    absh = 1 / rh;
 26end
 27absh = max(absh, hmin);
 28
 29f(:,1) = f0;
 30
 31% THE MAIN LOOP
 32
 33done = false;
 34while ~done
 35  
 36  % By default, hmin is a small number such that t+hmin is only slightly
 37  % different than t.  It might be 0 if t is 0.
 38  hmin = 16*eps(t);
 39  absh = min(hmax, max(hmin, absh));    % couldn't limit absh until new hmin
 40  h = absh;
 41  
 42  % Stretch the step if within 10% of tfinal-t.
 43  if 1.1*absh >= abs(tfinal - t)
 44    h = tfinal - t;
 45    absh = abs(h);
 46    done = true;
 47  end
 48  
 49  % LOOP FOR ADVANCING ONE STEP.
 50  nofailed = true;                      % no failed attempts
 51  while true
 52      hA = h * A;
 53      hB = h * B;
 54      f(:,2) = StCO_Dynamics_MEE(t+hA(1),y+f*hB(:,1), c, Thr, SI2CAN, kd);       %<--------------------------
 55      f(:,3) = StCO_Dynamics_MEE(t+hA(2),y+f*hB(:,2), c, Thr, SI2CAN, kd);       %<--------------------------
 56      f(:,4) = StCO_Dynamics_MEE(t+hA(3),y+f*hB(:,3), c, Thr, SI2CAN, kd);       %<--------------------------
 57      f(:,5) = StCO_Dynamics_MEE(t+hA(4),y+f*hB(:,4), c, Thr, SI2CAN, kd);       %<--------------------------
 58      f(:,6) = StCO_Dynamics_MEE(t+hA(5),y+f*hB(:,5), c, Thr, SI2CAN, kd);       %<--------------------------
 59      
 60      tnew = t + hA(6);
 61      if done
 62          tnew = tfinal;   % Hit end point exactly.
 63      end
 64%       h = tnew - t;      % Purify h.
 65      
 66      ynew = y + f*hB(:,6);
 67      f(:,7) = StCO_Dynamics_MEE(tnew,ynew, c, Thr, SI2CAN, kd);                 %<--------------------------
 68      % Estimate the error.
 69      err = absh * norm((f * E) ./ max(max(abs(y),abs(ynew)),threshold),inf);
 70      
 71      
 72      % Accept the solution only if the weighted error is no more than the
 73      % tolerance rtol.  Estimate an h that will yield an error of rtol on
 74      % the next step or the next try at taking this step, as the case may be,
 75      % and use 0.8 of this value to avoid failures.
 76      if err > rtol                       % Failed step
 77          if absh <= hmin
 78              
 79              t_out = 0;
 80              y_out = zeros(Neq,1);
 81%             sprintf('tolerance not satisfied');
 82              return;
 83          end
 84          
 85          if nofailed
 86              nofailed = false;
 87              absh = max(hmin, absh * max(0.1, 0.8*(rtol/err)^pow));
 88              
 89          else
 90              absh = max(hmin, 0.5 * absh);
 91          end
 92          h = absh;
 93          done = false;
 94      else
 95          
 96          break;
 97      end
 98  end
 99             
100  if done
101    
102    break;
103  end
104  
105  % If there were no failures compute a new h.
106  if nofailed
107    % Note that absh may shrink by 0.8, and that err may be 0.
108    temp = 1.25*(err/rtol)^pow;
109    if temp > 0.2
110      absh = absh / temp;
111    else
112      absh = 5.0*absh;
113    end
114  end
115  
116  % Advance the integration one step.
117  t = tnew;
118  y = ynew;
119  
120  f(:,1) = f(:,7);  % Already have f(tnew,ynew)
121 
122end
123t_out = tnew;
124y_out = ynew;
125
126end
